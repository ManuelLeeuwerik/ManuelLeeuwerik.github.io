model <- '
# direct effect
vaccine_standard  ~  b1_stand*risky_choices + b2_stand*lotto_tickets +  c_stand*income
vaccine_fast  ~  b1_fast*risky_choices + b2_fast*lotto_tickets +  c_fast*income
flu_vaccine ~ b1_flu*risky_choices + b2_flu*lotto_tickets +  c_flu*income

# mediator
risky_choices ~ a1 * income
lotto_tickets  ~ a2 * income

# direct effect
direct_stand := c_stand
direct_fast := c_fast
direct_flu := c_flu

# indirect effect (a*b)
indirect1_stand := a1*b1_stand
indirect2_stand := a2*b2_stand
indirect1_fast := a1*b1_fast
indirect2_fast := a2*b2_fast
indirect1_flu := a1*b1_flu

# total effect
total_stand := c_stand + (a1 * b1_stand) + (a2 * b2_stand) 
total_fast := c_fast + (a1 * b1_fast) + (a2 * b2_fast)
total_flu := c_flu + (a1 * b1_flu) + (a2 * b2_flu)

# percentage mediation
perc_stand := ((indirect1_stand + indirect2_stand) / total_stand)*100
perc_fast := ((indirect1_fast + indirect2_fast) / total_fast)*100
perc_flu := ((indirect1_flu + indirect2_flu) / total_flu)*100

risky_choices ~~ lotto_tickets
'
